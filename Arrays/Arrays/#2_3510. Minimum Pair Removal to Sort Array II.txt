Given an array nums, you can perform the following operation any number of times:

Select the adjacent pair with the minimum sum in nums. If multiple such pairs exist, choose the leftmost one.
Replace the pair with their sum.
Return the minimum number of operations needed to make the array non-decreasing.

An array is said to be non-decreasing if each element is greater than or equal to its previous element (if it exists).

 

Example 1:

Input: nums = [5,2,3,1]

Output: 2

Explanation:

The pair (3,1) has the minimum sum of 4. After replacement, nums = [5,2,4].
The pair (2,4) has the minimum sum of 6. After replacement, nums = [5,6].
The array nums became non-decreasing in two operations.

Example 2:

Input: nums = [1,2,2]

Output: 0

Explanation:

The array nums is already sorted.

 

Constraints:

1 <= nums.length <= 105
-109 <= nums[i] <= 109

Solution(#1) => O(n + logn)

class Solution {
public:
    int minimumPairRemoval(vector<int>& nums) {
        int n = nums.size();
        int badPairs = 0;
        set<pair<long long, int>> pairs;
        vector<int> prev(n), next(n);
        
        // Use a long long vector to prevent overflow when updating sums
        vector<long long> temp(n);

        for(int i = 0; i < n; i++) {
            temp[i] = nums[i]; // Store original values as long long
            prev[i] = i - 1;
            next[i] = i + 1;
        }

        for(int i = 0; i < n - 1; i++) {
            // Calculate sum safely using long long
            pairs.insert({temp[i] + temp[i+1], i});
            
            if(temp[i] > temp[i+1]) {
                badPairs++;
            }
        }

        int count = 0;
        while(badPairs > 0) {
            auto min_pair = pairs.begin();
            int first = min_pair->second;
            int second = next[first];

            int first_left = prev[first];
            int second_right = next[second];

            pairs.erase(pairs.begin());

            if(temp[first] > temp[second]) {
                badPairs--;
            }

            // Check neighbor relations before merging
            if(first_left != -1) {
                if(temp[first_left] > temp[first] && temp[first_left] <= temp[first] + temp[second]) badPairs--;
                else if(temp[first_left] <= temp[first] && temp[first_left] > temp[first] + temp[second]) badPairs++;
            }

            if(second_right != n) {
                if(temp[second_right] >= temp[second] && temp[second_right] < temp[first] + temp[second]) badPairs++;
                else if(temp[second_right] < temp[second] && temp[second_right] >= temp[first] + temp[second]) badPairs--;
            }

            // Update the set with new neighbor sums
            if(first_left != -1) {
                pairs.erase({temp[first_left] + temp[first], first_left});
                pairs.insert({temp[first_left] + temp[first] + temp[second], first_left});
            } 

            if(second_right != n) {
                pairs.erase({temp[second_right] + temp[second], second});
                pairs.insert({temp[first] + temp[second] + temp[second_right], first});
                prev[second_right] = first;
            } 

            next[first] = second_right; 
            temp[first] += temp[second]; // Now safe from overflow

            count++;
        }
        return count;
    }
};
