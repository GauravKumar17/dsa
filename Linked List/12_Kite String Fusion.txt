ListNode* addTwoNumbers(ListNode* l1, ListNode* l2, int k) {
    ListNode* temp = l1;
    ListNode* temp2 = l2;

    ListNode* dummy = new ListNode(-1);
    ListNode* temp3 = dummy;
    int carry = 0;
    while(temp || temp2 || carry){
         int sum = carry;
         if(temp){
            sum += temp->val;
            temp = temp->next;
         }
         if(temp2){
            sum += temp2->val;
            temp2 = temp2->next;
         }
            carry = sum/10;;
            sum = sum%10;
         
         int value = (sum *k )%10;
        dummy ->next = new ListNode(value);
        dummy = dummy->next;
    
    }
    return temp3->next;

}
