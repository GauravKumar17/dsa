#include<iostream>
#include<bits/stdc++.h>
using namespace std;

int main(){
    int n=5;
    vector<vector<int>> edgelist = {
        {0, 1},
        {0, 2},
        {1, 2},
        {1, 3},
        {2, 4},
        {3, 4}
    };

    //Matrix
    vector<vector<int>> adjMatrix(n+1,vector<int>(n+1,0));

    for(auto edge: edgelist){
        int u = edge[0];
        int v = edge[1];
        adjMatrix[u][v] = 1;
        adjMatrix[v][u] = 1;
    }

    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++){
            cout<<adjMatrix[i][j]<<" ";
        }
        cout<<endl;
    }

    //List
    vector<vector<int>> adjList(n);
    for(auto edge: edgelist){
        int u = edge[0];
        int v = edge[1];
        adjList[u].push_back(v);
        adjList[v].push_back(u);
    }

    for(int i=0;i<n;i++){
        cout<<i<<"-> ";
        for(auto edge: adjList[i]){
            cout<<edge<<" ";
        }
        cout<<endl;
    }

    //weighted adjacency list

    return 0;

}
