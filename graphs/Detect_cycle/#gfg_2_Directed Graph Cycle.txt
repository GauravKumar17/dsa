Given a Directed Graph with V vertices (Numbered from 0 to V-1) and E edges, check whether it contains any cycle or not.
The graph is represented as a 2D vector edges[][], where each entry edges[i] = [u, v] denotes an edge from vertex u to v.

Examples:

Input: V = 4, edges[][] = [[0, 1], [1, 2], [2, 0], [2, 3]]



Output: true
Explanation: The diagram clearly shows a cycle 0 → 1 → 2 → 0
Input: V = 4, edges[][] = [[0, 1], [0, 2], [1, 2], [2, 3]]


Output: false
Explanation: no cycle in the graph
Constraints:
1 ≤ V ≤ 105
0 ≤ E ≤ 105
0 ≤ edges[i][0], edges[i][1] < V

Solution(#DFS) =>

Intution:
Why normal DFS is not enough:-

In a directed graph, a node can be visited again through a different path that does not form a cycle.
So using only one visited array cannot reliably detect cycles.

Key Idea:-

1.A cycle exists only if a node is revisited while it is still in the current DFS path.

    To track this, we use two arrays:

    vis[] → marks nodes that have been visited at least once

    pathVis[] → marks nodes that are part of the current recursion path

2.vis[] prevents reprocessing nodes

3.pathVis[] detects back edges

4.A cycle is confirmed only when a node appears again in the same DFS path, not through some other completed path


class Solution {
  public:
    bool dfs(int i,int parent,vector<int>&vis,vector<vector<int>>& adj){
        vis[i] = 1;
        for(int x : adj[i]){
            if(vis[x] == 0){
                if(dfs(x,i,vis,adj)) return true;
            }
            else{
                if(x != parent){
                    return true;
                }
                
            }
        }
        
        return false;
       
    }
    bool isCycle(int V, vector<vector<int>>& edges) {
        vector<vector<int>>adj(V);
        
        for(int i=0;i<edges.size();i++){
            adj[edges[i][0]].push_back(edges[i][1]);
            adj[edges[i][1]].push_back(edges[i][0]);
        }
        
        vector<int>vis(V,0);
            
            for(int i=0;i<V;i++){
                bool res = false;
                
                if(vis[i] == 0){
                    res = dfs(i,-1,vis,adj);
                }
                
                if(res){
                    return true;
                }
                
            }
            return false;

            
        }
    
};



# Solution(#2) = > BFS (Using topological sort): [Kahn's Algo]

Intutuion: If after applying kahn's algorithm on a directed graph of N vertices 
            If our length of linear ordering (topo sort) is less than N, then the graph has cycle.

class Solution {
  public:
   
    bool isCycle(int V, vector<vector<int>>& edges) {
        
        vector<vector<int>>adj(V);
        vector<int>indegree(V,0),res;
        
        for(int i=0;i<edges.size();i++){
            adj[edges[i][0]].push_back(edges[i][1]);
            indegree[edges[i][1]]++;
        }
        
        queue<int>q;
        for(int i=0;i<indegree.size();i++){
            if(indegree[i] == 0){
                q.push(i);
            }
        }
        
        while(!q.empty()){
            int node = q.front();
            q.pop();
            res.push_back(node);
            
            for(int x: adj[node]){
                indegree[x]--;
                if(indegree[x] == 0){
                    q.push(x);
                }
            }
            
        }
        
        return (res.size() == V)? true : false;
    }
};
